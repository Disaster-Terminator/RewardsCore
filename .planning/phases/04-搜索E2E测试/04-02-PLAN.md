import pytest
from playwright.async_api import Page

class TestSearchQuota:
    """Tests that verify search quota tracking with authentication."""

    @pytest.mark.e2e
    @pytest.mark.requires_login
    @pytest.mark.timeout(120)
    async def test_search_increments_quota(self, page: Page, search_with_points_check):
        """
        Performing searches should increment search quota/points on rewards dashboard.
        Note: Actual points per search vary; we verify that quota count increases.
        """
        page, get_points = search_with_points_check

        # Navigate to Bing homepage
        await page.goto("https://www.bing.com", wait_until="domcontentloaded")

        # Record initial points/quota
        initial_points = await get_points()
        initial_quota = await self._get_search_quota_from_dashboard(page)

        # Perform one search
        await page.fill("input[name='q']", "test search quota increment")
        await page.keyboard.press("Enter")
        await page.wait_for_selector(".b_algo", timeout=10000)

        # Navigate back to rewards dashboard
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")

        # Verify quota increased by 1
        new_quota = await self._get_search_quota_from_dashboard(page)
        assert new_quota > initial_quota, f"Search quota did not increase: {initial_quota} -> {new_quota}"

    async def _get_search_quota_from_dashboard(self, page: Page) -> int:
        """Extract search quota from rewards dashboard."""
        # Quota might be shown as "0/10" or "3/10" for daily searches
        # Look for elements with data-ct containing "search" and numbers
        try:
            # Common pattern: progress bar text showing "X of Y"
            progress_text = await page.text_content(".progressContainer, [data-ct*='search']")
            import re
            match = re.search(r'(\d+)\s*(?:of|/)\s*(\d+)', progress_text)
            if match:
                current = int(match.group(1))
                return current
        except:
            pass
        return 0

    @pytest.mark.e2e
    @pytest.mark.requires_login
    @pytest.mark.timeout(120)
    async def test_multiple_searches_increase_quota(self, page: Page):
        """Multiple searches should accumulate until daily limit reached."""
        await page.goto("https://www.bing.com", wait_until="domcontentloaded")

        initial_quota = await self._get_search_quota_from_dashboard(page)

        # Perform 3 searches
        for i in range(3):
            term = f"test search number {i+1}"
            await page.fill("input[name='q']", term)
            await page.keyboard.press("Enter")
            await page.wait_for_selector(".b_algo", timeout=10000)
            await page.wait_for_timeout(2000)  # Wait for quota update

        # Check quota
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")
        new_quota = await self._get_search_quota_from_dashboard(page)
        assert new_quota >= initial_quota + 3, f"Expected at least +3 searches, got {new_quota - initial_quota}"

    @pytest.mark.e2e
    @pytest.mark.requires_login
    @pytest.mark.timeout(90)
    async def test_search_points_awarded(self, page: Page):
        """
        Verify that search points are awarded (may depend on account level/promotions).
        Note: Points awarding can be delayed; this test is informational.
        """
        pytest.skip("Points awarding is account-specific and may not be immediate; used as integration check only")
```

**Acceptance:** Search quota tracking verified (quota increments with each search).

import pytest
from playwright.async_api import Page

class TestSearchErrors:
    """Tests for search error conditions and rate limiting."""

    @pytest.mark.e2e
    @pytest.mark.requires_login
    @pytest.mark.timeout(60)
    async def test_rate_limit_not_triggered_by_normal_usage(self, page: Page):
        """
        Normal search pace (config wait_interval) should not trigger rate limiting.
        """
        # Perform searches with reasonable intervals (5-10s between)
        for i in range(3):
            await page.goto("https://www.bing.com", wait_until="domcontentloaded")
            await page.fill("input[name='q']", f"test rate limit {i}")
            await page.keyboard.press("Enter")
            await page.wait_for_selector(".b_algo", timeout=10000)
            # Wait for search quota to update
            await page.wait_for_timeout(5000)

        # Should not see rate limit message (no CAPTCHA or "too many requests")
        page_text = await page.text_content("body")
        assert "CAPTCHA" not in page_text, "Rate limit triggered unexpectedly"
        assert "too many requests" not in page_text.lower()

    @pytest.mark.e2e
    @pytest.mark.timeout(60)
    async def test_empty_search_query_handled(self, page: Page):
        """Submitting empty search should not crash or redirect."""
        await page.goto("https://www.bing.com", wait_until="domcontentloaded")

        search_box = await page.query_selector("input[name='q']")
        await search_box.focus()
        await page.keyboard.press("Enter")

        # Should stay on homepage or show validation message
        await page.wait_for_timeout(2000)
        # No crash - page still functional
        title = await page.title()
        assert "Bing" in title

    @pytest.mark.e2e
    @pytest.mark.timeout(60)
    async def test_special_characters_in_search(self, page: Page):
        """Search with special characters should not break."""
        special_terms = [
            "C++ tutorial",
            "Python 3.12 & async",
            "test@example.com",
            "100% complete",
            "price: $500"
        ]

        for term in special_terms:
            await page.goto("https://www.bing.com", wait_until="domcontentloaded")
            await page.fill("input[name='q']", term)
            await page.keyboard.press("Enter")
            try:
                await page.wait_for_selector(".b_algo", timeout=10000)
                results = await page.query_selector(".b_algo")
                assert results is not None, f"No results for: {term}"
            except Exception as e:
                pytest.fail(f"Search failed for '{term}': {e}")
```

**Acceptance:** Error scenarios handled gracefully; no crashes or unexpected rate limits.
