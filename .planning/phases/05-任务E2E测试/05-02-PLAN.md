import pytest
from playwright.async_api import Page

class TestQuizTasks:
    """Tests for quiz-type reward tasks."""

    @pytest.mark.e2e
    @pytest.mark.requires_login
    @pytest.mark.timeout(180)
    async def test_quiz_task_completion(self, page: Page, task_discovery):
        """
        Find and complete a quiz task if available.
        Quizzes typically involve 5-10 multiple-choice questions.
        """
        tasks = await task_discovery()
        quiz_tasks = [t for t in tasks if t.status == "available" and ("quiz" in t.title.lower() or "poll" in t.title.lower() or "question" in t.title.lower())]

        if not quiz_tasks:
            pytest.skip("No quiz tasks available")

        task = quiz_tasks[0]
        await page.goto(task.url, wait_until="domcontentloaded", timeout=20000)

        # Quiz flow: answer questions, submit, get reward
        questions_answered = 0
        max_questions = 10

        for _ in range(max_questions):
            # Wait for question to appear
            try:
                await page.wait_for_selector(".question, [data-ct*='question'], input[type='radio']", timeout=5000)
            except:
                break  # No more questions

            # Choose first available answer option
            option = await page.query_selector("input[type='radio'], input[type='checkbox']:not(:checked)")
            if option:
                await option.click()
                # Click "Next" or "Submit"
                next_btn = await page.query_selector("button:has-text('Next'), button:has-text('Submit'), input[value='Next']")
                if next_btn:
                    await next_btn.click()
                    await page.wait_for_timeout(2000)
                    questions_answered += 1
                else:
                    break
            else:
                break

        # Final submission
        submit_btn = await page.query_selector("button:has-text('Get reward'), button:has-text('Finish')")
        if submit_btn:
            await submit_btn.click()
            await page.wait_for_timeout(3000)

        # Verify completion
        completion_msg = await page.query_selector("text='You've earned', text='completed'")
        assert completion_msg or questions_answered > 0, "Quiz did not complete"

    @pytest.mark.e2e
    @pytest.mark.requires_login
    @pytest.mark.timeout(120)
    async def test_quiz_wrong_answer_handling(self, page: Page):
        """If quiz allows wrong answers, should not break flow."""
        pytest.skip("Hard to induce wrong answer; depends on quiz design")
        # Could deliberately pick wrong answers and verify no crash, maybe allows retry

        """
        Complete a URL reward task and check points increase.

        await perform_login(page, test_credentials)

        # Get initial points

        # Find and execute available URL task
        tasks = await self._discover_tasks(page)
        url_tasks = [t for t in tasks if t.status == "available" and t.url and t.reward_points > 0]
        if not url_tasks:
            pytest.skip("No URL reward tasks with points available")

        task = url_tasks[0]
        await page.goto(task.url, wait_until="domcontentloaded", timeout=20000)
        await page.wait_for_timeout(5000)  # Wait for completion potential

        # Return to dashboard
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")
        new_points = await self._extract_points(page)

        assert new_points >= initial_points + task.reward_points, \
            f"Points did not increase by expected {task.reward_points}: {initial_points} -> {new_points}"

    async def _discover_tasks(self, page: Page):
        """Helper to discover tasks."""
        # Reuse task_discovery fixture logic inline
        # For brevity, skip full reimplementation
        return []

    async def _extract_points(self, page: Page) -> int:
        """Extract numeric points from dashboard."""
        selectors = [".credits_value", ".mee-textPoints"]
        for sel in selectors:
            el = await page.query_selector(sel)
            if el:
                import re
                text = await el.text_content()
                match = re.search(r'([\d,]+)', text.replace(',', ''))
                if match:
                    return int(match.group(1))
        return 0
