import pytest
import time
from pathlib import Path
from typing import Dict, Any

# Session-scoped profiler
@pytest.fixture(scope="session", autouse=True)
def smoke_profiler(request) -> Dict[str, Any]:
    """Track overall smoke suite performance metrics."""
    profiler = {
        "start": time.perf_counter(),
        "tests": [],
        "session_id": int(time.time())
    }
    yield profiler
    total = time.perf_counter() - profiler["start"]
    _print_smoke_summary(profiler, total)

def _print_smoke_summary(profiler: Dict[str, Any], total: float) -> None:
    """Print formatted summary at suite end and save profile."""
    from tests.e2e.helpers.environment import is_ci_environment

    ci = is_ci_environment()
    threshold = 45 if ci else 30

    print(f"\n{'='*60}")
    print(f"SMOKE SUITE SUMMARY (CI: {ci})")
    print(f"{'='*60}")
    for t in profiler["tests"]:
        print(f"  {t['name']}: {t['duration']:.3f}s")
    print(f"{'='*60}")
    print(f"Total: {total:.3f}s (threshold: {threshold}s)")
    if total > threshold:
        print(f"⚠️  WARNING: Suite exceeded performance threshold!")
    print(f"{'='*60}\n")

    # Save profile to disk
    profile_dir = Path("logs") / "e2e" / "smoke"
    profile_dir.mkdir(parents=True, exist_ok=True)
    profile_path = profile_dir / "profile.json"
    profile = {
        "total_seconds": total,
        "passed": total <= threshold,
        "threshold": threshold,
        "ci": ci,
        "tests": profiler["tests"],
        "session_id": profiler.get("session_id")
    }
    profile_path.write_text(json.dumps(profile, indent=2))

@pytest.fixture
def track_smoke_duration(request, smoke_profiler):
    """Record execution time of each smoke test."""

    yield
    duration = time.perf_counter() - start
    smoke_profiler["tests"].append({
        "name": request.node.name,
        "duration": duration
    })

### Task 2.1.2: Enhance Environment Detection (Two-Stage)

Extend `tests/e2e/helpers/environment.py`:

```python
from typing import Tuple

def check_playwright_installed() -> Tuple[bool, str]:
    """
    Stage 1: Static check for Playwright driver and browsers.
    Fast, doesn't launch browser.
    """
    try:
        from playwright._impl._driver import compute_driver_executable
        compute_driver_executable()
        return True, ""
    except Exception as e:
        return False, f"Playwright drivers not found: {e}"

def check_browser_binary() -> Tuple[bool, str]:
    """
    Stage 2: Dynamic check - try launching Chromium headless.
    More expensive but validates actual browser availability.
    """
    import asyncio
    from playwright.async_api import async_playwright

    async def _check():
        try:
            async with async_playwright() as p:
                browser = await p.chromium.launch(headless=True)
                await browser.close()
                return True, ""
        except Exception as e:
            return False, str(e)

    return asyncio.run(_check())

def require_smoke_environment():
    """
    Validate smoke test environment.
    Skip entire suite with actionable message if dependencies missing.
    """
    ok, msg = check_playwright_installed()
    if not ok:
        pytest.skip(f"[SKIP] Playwright not installed. Run: playwright install chromium. Details: {msg}")

    ok, msg = check_browser_binary()
    if not ok:
        pytest.skip(f"[SKIP] Chromium not available. Run: playwright install chromium. Details: {msg}")
```

**Acceptance:** Environment checks provide clear, actionable skip messages; detected missing dependencies correctly.

"""Bing homepage health checks."""
import pytest
from playwright.async_api import Page, expect

pytestmark = [pytest.mark.smoke, pytest.mark.no_login]

class TestBingHealth:
    """Verify Bing is accessible and functional."""

    async def test_navigate_to_bing_homepage(self, page: Page, track_smoke_duration):
        """Navigate to Bing and verify homepage loads."""
        await page.goto(
            "https://www.bing.com",
            wait_until="domcontentloaded",
            timeout=15000
        )
        title = await page.title()
        assert "Bing" in title, f"Unexpected page title: {title}"

    async def test_search_input_present_and_interactable(self, page: Page, track_smoke_duration):
        """Verify search input is visible and can receive text."""
        await page.goto(
            "https://www.bing.com",
            wait_until="domcontentloaded",
            timeout=15000
        )

        # Use stable selector: input[name='q'] is Bing's standard search box
        search_input = page.locator("input[name='q']")
        await expect(search_input).to_be_visible(timeout=10000)

        # Test interaction (fill + submit)
        await search_input.fill("playwright test")
        await page.keyboard.press("Enter")

        # Verify navigation to search results
        await page.wait_for_url(lambda url: "search?" in url, timeout=10000)
```

"""Flakiness tracking utilities for smoke tests."""
import json
import time
from pathlib import Path
from typing import Dict, List

FLAKINESS_DB = Path("logs") / "e2e" / "flakiness.json"

def record_outcome(test_name: str, passed: bool, duration: float):
    """Record test outcome for flakiness analysis."""
    entry = {
        "test": test_name,
        "passed": passed,
        "duration": duration,
        "timestamp": time.time()
    }
    # Append to JSONL file for simplicity
    with FLAKINESS_DB.open("a") as f:
        f.write(json.dumps(entry) + "\n")

def get_flakiness_report(window: int = 5) -> Dict[str, Dict]:
    """
    Analyze recent outcomes and return flakiness metrics.
    Window: number of most recent runs to analyze.
    """
    if not FLAKINESS_DB.exists():
        return {}

    lines = FLAKINESS_DB.read_text().strip().split("\n")
    recent = [json.loads(line) for line in lines[-window*100:]]  # last N entries

    # Group by test
    by_test: Dict[str, List[Dict]] = {}
    for entry in recent:
        by_test.setdefault(entry["test"], []).append(entry)

    # Compute pass rates
    report = {}
    for test, entries in by_test.items():
        recent_n = entries[:window]
        total = len(recent_n)
        passed = sum(1 for e in recent_n if e["passed"])
        report[test] = {
            "pass_rate": passed / total if total > 0 else 1.0,
            "total_runs": total,
            "passed": passed,
            "flaky": passed / total < 0.95 and total >= 3
        }

    return report
```

## Notes

- **Selector Stability**: `input[name='q']` is Bing's standard and more stable than CSS classes. Avoid `.b_algo` for critical assertions; prefer text-based checks (e.g., page.get_by_text()).
- **Wait Strategy**: Use `wait_until="domcontentloaded"` for fast navigation. Reserve full `load` for when you need all resources.
- **Timeouts**: Aggressive but reasonable (10-15s) to keep suite fast. Phase 1 fixtures should already set reasonable defaults.
- **Performance**: Use `time.perf_counter()` (monotonic) not `time.time()` for precision and reliability.
- **Environment Detection**: Two-stage approach catches both missing driver (static) and missing browser binary (dynamic) with clear, actionable messages.
- **Flakiness Prevention**: Use `expect()` assertions (auto-wait) rather than manual checks. Avoid `time.sleep()`.
- **Smoke Philosophy**: These tests are environment guardians. Fail fast, fail clearly. No credentials, no external dependencies beyond Bing accessibility.
