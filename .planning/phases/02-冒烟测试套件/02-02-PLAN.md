### Task 2.2.1: Search Execution Test

Create `tests/e2e/smoke/test_search_execution.py`:

```python
import pytest

class TestSearchExecution:
    """Smoke tests for search functionality."""

    @pytest.mark.smoke
    @pytest.mark.timeout(30)
    async def test_bing_search_execution(self, page):
        """Execute a search and verify results appear."""
        await page.goto("https://www.bing.com", wait_until="domcontentloaded")
        # Accept cookies if present
        accept_button = await page.query_selector("button#bnp_btn_accept")
        if accept_button:
            await accept_button.click()

        search_box = await page.query_selector("input[name='q']")
        assert search_box is not None, "Search box not found"

        query = "weather today"
        await search_box.fill(query)
        await page.keyboard.press("Enter")

        # Wait for results page
        await page.wait_for_url(lambda url: "search?" in url, timeout=10000)

        # Verify results container exists
        results = await page.query_selector(".b_algo")
        assert results is not None, "No search results found"

    @pytest.mark.smoke
    @pytest.mark.timeout(30)
    async def test_search_quota_increases(self, page):
        """Verify search count increments on Bing rewards dashboard."""
        await page.goto("https://www.bing.com", wait_until="domcontentloaded")
        # Perform a search
        await page.fill("input[name='q']", "test query")
        await page.keyboard.press("Enter")
        await page.wait_for_selector(".b_algo", timeout=10000)

        # Navigate to rewards dashboard
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")
        # Check for search progress indicator (may depend on account state)
        progress_element = await page.query_selector("[data-ct*='search']")
        # In no-login mode, this may not be visible; that's okay
        if progress_element:
            assert "progress" in progress_element.inner_text.lower() or "search" in progress_element.inner_text.lower()
```

**Acceptance:** Search execution tests pass with or without authentication.

### Task 2.2.3: Performance Profiling & Optimization

Create `tests/e2e/smoke/conftest.py`:

```python
import pytest
import time
import json
import os

@pytest.fixture(scope="session", autouse=True)
def profile_smoke_suite():
    """Record start/end times for the smoke suite."""
    start = time.time()
    yield
    end = time.time()
    total = end - start

    logs_dir = os.path.join(os.getcwd(), "logs", "e2e", "smoke")
    os.makedirs(logs_dir, exist_ok=True)
    profile_path = os.path.join(logs_dir, "profile.json")
    with open(profile_path, "w") as f:
        json.dump({"total_seconds": total, "pass": total < 30}, f, indent=2)

    if total >= 30:
        pytest.fail(f"Smoke suite took {total:.2f}s, exceeds 30s limit")

# Ensure parallel execution for speed
def pytest_configure(config):
    if not config.option.numprocesses:
        config.option.numprocesses = "auto"
```
