### Task 2.1.2: Basic Navigation Test

Create `tests/e2e/smoke/test_basic_navigation.py`:

```python
import pytest

class TestBasicNavigation:
    """Smoke tests for basic web navigation."""

    @pytest.mark.smoke
    @pytest.mark.timeout(30)
    async def test_navigate_to_bing(self, page):
        """Navigate to Bing homepage and verify title."""
        await page.goto("https://www.bing.com", wait_until="domcontentloaded")
        title = await page.title()
        assert "Bing" in title

    @pytest.mark.smoke
    @pytest.mark.timeout(30)
    async def test_bing_search_input_visible(self, page):
        """Verify search input is present and interactable."""
        await page.goto("https://www.bing.com", wait_until="domcontentloaded")
        search_box = await page.query_selector("input[name='q']")
        assert search_box is not None
        await search_box.fill("test")
        await page.keyboard.press("Enter")
        # Wait for results
        await page.wait_for_selector(".b_algo", timeout=10000)
```

**Acceptance:** Navigation and search input tests pass reliably.
