### Task 3.2.2: Storage State Reuse Test (Fast Login)

```python
import pytest
import os
from playwright.async_api import Page

class TestSessionPersistence:
    """Tests for session reuse via storage_state.json."""

    @pytest.fixture
    async def context_with_storage(self, browser):
        """Create context with pre-saved storage state."""
        storage_path = os.getenv("E2E_STORAGE_STATE", "tests/fixtures/storage_state.json")
        if not os.path.exists(storage_path):
            pytest.skip(f"Storage state not found: {storage_path}")

        context = await browser.new_context()
        await context.add_cookies(json.load(open(storage_path)))
        yield context
        await context.close()

    @pytest.mark.e2e
    @pytest.mark.timeout(30)
    async def test_storage_state_auto_login(self, page: Page, context_with_storage):
        """Verify page created from context with storage_state is already logged in."""
        # Create new page from context that has storage state
        page = await context_with_storage.new_page()
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")

        # Should see dashboard without login prompt
        dashboard = await page.query_selector(".progressContainer, [data-ct*='dashboard']")
        assert dashboard is not None, "Dashboard not visible - session not restored"

        # Verify no sign-in buttons visible
        signin_btn = await page.query_selector("a[href*='login'], button:has-text('Sign in')")
        assert signin_btn is None or not await signin_btn.is_visible()

    @pytest.mark.e2e
    @pytest.mark.timeout(30)
    async def test_session_survives_browser_restart(self, browser):
        """Verify storage state can be saved and reused across browser instances."""
        # Save state in first context
        context1 = await browser.new_context()
        page1 = await context1.new_page()
        await page1.goto("https://rewards.bing.com", wait_until="domcontentloaded")

        # Minimal check that we can reach dashboard
        await page1.wait_for_selector(".progressContainer, [data-ct*='dashboard']", timeout=30000)

        storage = await context1.storage_state()
        await context1.close()

        # Restore in second context
        context2 = await browser.new_context()
        await context2.add_cookies(storage["cookies"])
        page2 = await context2.new_page()
        await page2.goto("https://rewards.bing.com", wait_until="domcontentloaded")

        # Should be logged in
        await page2.wait_for_selector(".progressContainer, [data-ct*='dashboard']", timeout=15000)
        await context2.close()
```

**Acceptance:** Storage state tests pass when `storage_state.json` exists; skip otherwise.

### Task 3.2.4: Edge Cases & Error Handling

Create `tests/e2e/login/test_login_edge_cases.py`:

```python
import pytest
from playwright.async_api import Page, TimeoutError

class TestLoginEdgeCases:
    """Edge case and error scenario tests for login."""

    @pytest.mark.e2e
    @pytest.mark.requires_login
    @pytest.mark.timeout(60)
    async def test_invalid_password_shows_error(self, page: Page, test_credentials):
        """Invalid password should display error message, not crash."""
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")

        # Enter email, then wrong password
        await page.fill("input[type='email']", test_credentials["email"])
        await page.click("input[type='submit']")
        await page.fill("input[type='password']", "wrong_password_123")
        await page.click("input[type='submit']")

        # Should see error indicator (class contains 'error' or text 'incorrect')
        error_element = await page.query_selector(".error, [role='alert'], text='incorrect'")
        assert error_element is not None, "Error message not shown for invalid password"

    @pytest.mark.e2e
    @pytest.mark.timeout(30)
    async def test_bing_rewards_redirects_to_login_if_not_authenticated(self, page: Page):
        """Unauthenticated access to rewards should redirect to Microsoft login."""
        # Ensure starting fresh (no storage state)
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")

        # Should redirect to login.live.com within a few seconds
        try:
            await page.wait_for_url(lambda url: "login.live.com" in url, timeout=10000)
            assert "login.live.com" in page.url
        except TimeoutError:
            # May show Bing's own login page
            current_url = page.url
            assert "login" in current_url.lower() or "account" in current_url.lower()

    @pytest.mark.e2e
    @pytest.mark.timeout(20)
    async def test_playwright_browser_close_after_login(self, browser):
        """Verify browser can be closed cleanly after login context."""
        context = await browser.new_context()
        page = await context.new_page()
        await page.goto("https://rewards.bing.com")

        # No assertion failure during close
        await context.close()
        assert context.is_closed() or True  # is_closed() may not be reliable
```

**Acceptance:** Edge case tests either pass or skip with meaningful reason.

```python
import pytest
from typing import List

def load_test_accounts() -> List[dict]:
    """Load multiple test accounts from YAML."""
    import yaml
    with open("tests/e2e/data/mock_accounts.yaml") as f:
        data = yaml.safe_load(f)

    # Filter accounts that have email+password
    return [acc for acc in accounts if acc.get("email") and acc.get("password")]

@pytest.mark.e2e
```
