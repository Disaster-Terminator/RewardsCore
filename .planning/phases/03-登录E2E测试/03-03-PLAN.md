from playwright.async_api import Page

    """Tests for account security scenarios (lockout, blocked)."""

    @pytest.mark.e2e
    @pytest.mark.slow  # May require manual intervention or waiting
    @pytest.mark.timeout(180)
    async def test_account_locked_detection(self, page: Page, test_credentials):
        """Test that lockout state is properly detected and reported."""
        # This test assumes account may be locked from too many attempts
        # Or simulate lockout via invalid password attempts (careful in CI)

        # Attempt multiple failed logins (don't actually lock a real account)
        # Instead, check for lockout indicator on rewards page if already locked

        from tests.e2e.helpers.account_health import check_account_health

        if health["reason"] == "account_locked":
            pytest.skip(f"Account already locked: {health.get('details', '')}")
        else:
            pytest.skip("Account not locked - test requires pre-locked account")

    @pytest.mark.e2e
    @pytest.mark.timeout(60)
    async def test_lockout_message_displayed(self, page: Page):
        """When redirected to rewards with locked account, error message visible."""
        # Navigate to rewards; if account locked, should see message
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")

        # Check for lockout-related text
        lockout_texts = [
            "account is locked",
            "account has been blocked",
            "suspicious activity",
            "verify your identity"
        ]

        page_text = await page.text_content("body")
        found = any(text.lower() in page_text.lower() for text in lockout_texts)

        if found:
            assert True, "Lockout message detected (expected in this test context)"
        else:
            pytest.skip("No lockout message - account not locked")
```

**Acceptance:** Tests correctly detect and report lockout conditions without causing actual locks.

import pytest
from playwright.async_api import Page

class TestInvalidCredentials:
    """Tests for wrong password, non-existent account, etc."""

    @pytest.mark.e2e
    @pytest.mark.timeout(60)
    async def test_wrong_password_error_message(self, page: Page, test_credentials):
        """Wrong password should show specific error without page crash."""
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")

        # Enter correct email, wrong password
        await page.fill("input[type='email']", test_credentials["email"])
        await page.click("input[type='submit']")
        await page.fill("input[type='password']", "wrong_password_xyz123")
        await page.click("input[type='submit']")

        # Look for error message (class or text)
        error_selectors = [
            ".error", "[role='alert']", "text='incorrect'", "text='wrong'",
            "text='Incorrect password'", "text="]
        error_found = False
        for selector in error_selectors:
            try:
                el = await page.query_selector(selector)
                if el:
                    text = await el.text_content()
                    if text and ("incorrect" in text.lower() or "wrong" in text.lower() or "error" in text.lower()):
                        error_found = True
                        break
            except:
                continue

        assert error_found, "No error message displayed for wrong password"

    @pytest.mark.e2e
    @pytest.mark.timeout(60)
    async def test_enter_key_submits_login_form(self, page: Page, test_credentials):
        """Pressing Enter in password field should submit login."""
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")

        await page.fill("input[type='email']", test_credentials["email"])
        await page.click("input[type='submit']")
        await page.fill("input[type='password']", test_credentials["password"])
        await page.keyboard.press("Enter")

        # Should reach dashboard or 2FA or error, not hang
        try:
            await page.wait_for_selector(".progressContainer, [role='alert'], input[name='otp']", timeout=20000)
        except Exception as e:
            pytest.fail(f"Login form submission via Enter key failed: {e}")

    @pytest.mark.e2e
    @pytest.mark.timeout(60)
    async def test_recovery_options_available_on_forgot_password(self, page: Page):
        """Verify 'Forgot password' link navigates to recovery page."""
        await page.goto("https://rewards.bing.com", wait_until="domcontentloaded")

        # Look for forgot password link
        forgot_link = await page.query_selector("a:has-text('Forgot'), a:has-text('Can't access'), a[href*='password']")
        if forgot_link:
            # Verify link is visible and has expected href
            href = await forgot_link.get_attribute("href")
            assert href, "Forgot password link has no href"
            assert "password" in href.lower() or "recover" in href.lower()
        else:
            pytest.skip("Forgot password link not found (may vary by region/page)")
```

    logs_dir = os.path.join(os.getcwd(), "logs", "e2e", "login_diagnostics")

    filename = f"{test_name}_{timestamp}.json"
    path = os.path.join(logs_dir, filename)

        "test": test_name,
        "timestamp": timestamp,
        **kwargs

@pytest.hookimpl(tryfirst=True, hookwrapper=True)
def pytest_runtest_makereport(item, call):
    outcome = yield
    report = outcome.get_result()

    if report.when == "call" and report.failed:
        # Save diagnostic data
        page = item.funcargs.get("page")
        if page:
            test_name = item.nodeid.replace("::", "_").replace("/", "_")
            screenshot_path = f"logs/e2e/screenshots/{test_name}.png"
