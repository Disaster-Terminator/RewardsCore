### Task 6.1.1: Smoke Test Workflow (On Push & PR)

Create `.github/workflows/e2e-smoke.yml`:

```yaml
name: E2E Smoke Tests

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main, develop ]

jobs:
  smoke-tests:
    runs-on: ubuntu-latest
    timeout-minutes: 15

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Install system dependencies
        run: |
          sudo apt-get update
          sudo apt-get install -y \
            libnss3 \
            libatk1.0-0 \
            libatk-bridge2.0-0 \
            libcups2 \
            libxcomposite1 \
            libxdamage1 \
            libxrandr2 \
            libgbm1 \
            libasound2 \
            libpangocairo-1.0-0 \
            libcairo2 \
            libpango-1.0-0 \
            libgtk-3.0

      - name: Install Python dependencies
        run: |
          pip install --upgrade pip
          pip install -e ".[dev]"

      - name: Install Playwright browsers
        run: |
          playwright install chromium
          playwright install-deps chromium

      - name: Create env file from secrets
        run: |
          echo "MS_REWARDS_E2E_EMAIL=${{ secrets.MS_REWARDS_E2E_EMAIL }}" >> .env.test
          echo "MS_REWARDS_E2E_PASSWORD=${{ secrets.MS_REWARDS_E2E_PASSWORD }}" >> .env.test
          echo "MS_REWARDS_E2E_TOTP_SECRET=${{ secrets.MS_REWARDS_E2E_TOTP_SECRET }}" >> .env.test
          echo "E2E_STORAGE_STATE=tests/fixtures/storage_state.json" >> .env.test
          cat .env.test

      - name: Upload storage state artifact (if exists)
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: storage-state
          path: tests/fixtures/storage_state.json
          retention-days: 7

      - name: Run smoke tests (no-login only)
        run: |
          pytest tests/e2e/smoke/ -v --tb=short -m "no_login or smoke"
        env:
          E2E_HEADLESS: true

      - name: Upload test results (JUnit)
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: pytest-results-s smoke
          path: |
            pytest.xml
            logs/e2e/**/*.json
          retention-days: 7

      - name: Upload screenshots on failure
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: screenshots-smoke
          path: logs/e2e/screenshots/
          retention-days: 7

      - name: Upload logs
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: logs-smoke
          path: logs/e2e/
          retention-days: 7
```

**Acceptance:** Workflow triggers on push/PR, completes in <15 min, artifacts uploaded on failure.

### Task 6.1.2: Full E2E Workflow (Scheduled Daily)

Create `.github/workflows/e2e-full.yml`:

```yaml
name: E2E Full Tests

on:
  schedule:
    - cron: '0 18 * * *'  # Daily at 6 PM UTC
  workflow_dispatch:     # Manual trigger

jobs:
  e2e-tests:
    runs-on: ubuntu-latest
    timeout-minutes: 30

    steps:
      - name: Checkout code
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.11'

      - name: Install system dependencies
        run: |
          sudo apt-get update
          sudo apt-get install -y \
            libnss3 libatk1.0-0 libatk-bridge2.0-0 libcups2 \
            libxcomposite1 libxdamage1 libxrandr2 libgbm1 \
            libasound2 libpangocairo-1.0-0 libcairo2 libpango-1.0-0 libgtk-3.0

      - name: Install Python dependencies
        run: |
          pip install --upgrade pip
          pip install -e ".[dev]"

      - name: Install Playwright browsers
        run: |
          playwright install chromium
          playwright install-deps chromium

      - name: Validate secrets configured
        run: |
          if [ -z "${{ secrets.MS_REWARDS_E2E_EMAIL }}" ]; then
            echo "ERROR: MS_REWARDS_E2E_EMAIL secret not set"
            exit 1
          fi
          if [ -z "${{ secrets.MS_REWARDS_E2E_PASSWORD }}" ]; then
            echo "ERROR: MS_REWARDS_E2E_PASSWORD secret not set"
            exit 1
          fi

      - name: Create env file from secrets
        run: |
          echo "MS_REWARDS_E2E_EMAIL=${{ secrets.MS_REWARDS_E2E_EMAIL }}" >> .env.test
          echo "MS_REWARDS_E2E_PASSWORD=${{ secrets.MS_REWARDS_E2E_PASSWORD }}" >> .env.test
          echo "MS_REWARDS_E2E_TOTP_SECRET=${{ secrets.MS_REWARDS_E2E_TOTP_SECRET }}" >> .env.test
          echo "E2E_STORAGE_STATE=tests/fixtures/storage_state.json" >> .env.test

      - name: Generate storage state (first run only)
        if: vars.need_storage_state != 'false'  # Configurable via repository variable
        run: |
          python scripts/save_storage_state.py
        env:
          E2E_HEADLESS: true
        continue-on-error: true  # Storage state generation may fail; not fatal

      - name: Run E2E tests (no-login first)
        run: |
          pytest tests/e2e/ -v --tb=short -m "no_login" --junitxml=pytest-no-login.xml
        env:
          E2E_HEADLESS: true

      - name: Run E2E tests (with-login)
        run: |
          pytest tests/e2e/ -v --tb=short -m "requires_login" --junitxml=pytest-with-login.xml
        env:
          E2E_HEADLESS: true

      - name: Upload test results
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: pytest-results-full
          path: |
            pytest-*.xml
            logs/e2e/**/*.json
            logs/daily_reports/
          retention-days: 30

      - name: Upload screenshots
        if: failure()
        uses: actions/upload-artifact@v4
        with:
          name: screenshots-full
          path: logs/e2e/screenshots/
          retention-days: 30

      - name: Upload diagnostic reports
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: diagnostics-full
          path: logs/diagnosis/**/*
          retention-days: 30

      - name: Check for flaky tests
        if: always()
        run: |
          # Could implement flakiness detection via pytest-rerunfailures or custom script
          echo "Flakiness check not implemented yet"
```

**Acceptance:** Full E2E workflow runs daily or on-demand, completes within 30 min, captures artifacts.

```markdown
# GitHub Actions Secrets for E2E Tests

The E2E workflows require the following repository secrets (Settings > Secrets and variables > Actions).

## Required Secrets

| Secret Name | Description | Format |
|-------------|-------------|--------|
| `MS_REWARDS_E2E_EMAIL` | Test Microsoft account email | `user@outlook.com` |
| `MS_REWARDS_E2E_PASSWORD` | Test account password | Plain text (no special escaping) |
| `MS_REWARDS_E2E_TOTP_SECRET` | Optional: 2FA TOTP base32 secret | `JBSWY3DPEHPK3PXP` (from Microsoft Authenticator) |

## Setting Up Secrets

1. Go to repository Settings
2. Navigate to **Secrets and variables** → **Actions**
3. Click **New repository secret**
4. Add each secret with Name and Value
5. Save

## Security Notes

- **Never** commit `.env.test` or `storage_state.json` to repository
- Use dedicated test accounts, not personal main accounts
- Rotate credentials if accidentally exposed
- Consider using GitHub Environments for staging/prod separation

## CI Variables (Optional)

Also define repository variables (Settings → Actions → Variables):

| Variable | Description | Default |
|----------|-------------|---------|
| `E2E_HEADLESS` | Run browser headless? | `true` |
| `E2E_STORAGE_STATE_GENERATE` | Generate storage_state on first run? | `true` |
| `E2E_SLACK_WEBHOOK` | Optional: Notification webhook URL | (empty) |

## Testing CI Configuration

After setting secrets, manually trigger workflow:

1. Go to repository **Actions** tab
2. Select "E2E Smoke Tests" or "E2E Full Tests"
3. Click **Run workflow** → **Run workflow**
4. Monitor logs for failures

## Troubleshooting

- **Secret not found**: Verify secret name matches exactly (case-sensitive)
- **Permission denied**: Secrets are encrypted; ensure repository has write access
- **Browser install fails**: Check `playwright install-deps` step has sudo privileges
```

### Task 6.1.4: Workflow Optimization & Caching

Enhance workflows with caching:

Add to `.github/workflows/e2e-smoke.yml` and `e2e-full.yml`:

```yaml
      - name: Cache Playwright browsers
        uses: actions/cache@v4
        with:
          path: |
            ~/.cache/ms-playwright
            /usr/bin/chromium
          key: ${{ runner.os }}-playwright-${{ hashFiles('**/requirements*.txt') }}
          restore-keys: |
            ${{ runner.os }}-playwright-

      - name: Cache pip dependencies
        uses: actions/cache@v4
        with:
          path: ~/.cache/pip
          key: ${{ runner.os }}-pip-${{ hashFiles('**/pyproject.toml') }}
          restore-keys: |
            ${{ runner.os }}-pip-
```

**Acceptance:** CI runs with caching enabled; first run slower, subsequent faster.

## Notes

- **CI Runtime Costs**: Ubuntu runners are free for public repos, limited minutes for private. Monitor usage.
- **Secret Management**: Use repository secrets, never environment variables in workflow YAML.
- **Headless Mode**: Always `E2E_HEADLESS: true` in CI. Local development can use headed for debugging.
- **Flaky Tests**: If CI shows flakiness, add `@pytest.mark.flaky(reruns=3)` or fix root cause.
- **Browser Installs**: `playwright install-deps` needs sudo; GitHub ubuntu-latest provides this.
